PROGRAM       : STAT*

STAT          : RETURN or
                DECL or
                IF

RETURN        : return EXPR;

DECL          : decl id = EXPR;

EXPR          : BINOP or
                int or
                id

BINOP         : EXPR + EXPR or      prec = 1
                EXPR - EXPR or      prec = 1
                EXPR * EXPR or      prec = 2
                EXPR / EXPR         prec = 2

IF            : if EXPR { STAT* }
