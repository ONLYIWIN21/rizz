PROGRAM       : STAT*

STAT          : RETURN or
                DECL or
                IF or
                ASSIGN or
                FUNC or
                COMMENT
                FOR

RETURN        : return EXPR;

DECL          : decl id = EXPR;

ASSIGN        : id = EXPR;

COMMENT       : // .*

FOR           : for DECL EXPR; ASSIGN { STAT* }

EXPR          : OP or
                int or
                id

OP            : EXPR || EXPR or     prec = 0
                EXPR && EXPR or     prec = 0
                EXPR == EXPR or     prec = 1
                EXPR + EXPR or      prec = 2
                EXPR - EXPR or      prec = 2
                EXPR * EXPR or      prec = 3
                EXPR / EXPR or      prec = 3
                EXPR % EXPR or      prec = 3
                ! EXPR              prec = 4
                ( EXPR )            prec = 5

IF            : if EXPR { STAT* }

FUNC          : func id ( id* ) { STAT* }
